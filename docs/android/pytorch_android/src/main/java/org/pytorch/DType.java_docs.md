  UINT8(1),
  /** Code for dtype torch.int8. {@link Tensor#dtype()} */
  INT8(2),

  INT32(3),
  /** Code for dtype torch.float32. {@link Tensor#dtype()} */

  /** Code for dtype torch.int64. {@link Tensor#dtype()} */
  INT64(5),
  /** Code for dtype torch.float64. {@link Tensor#dtype()} */
  FLOAT64(6),
