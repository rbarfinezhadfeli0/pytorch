BackendRegistrar::BackendRegistrar(ContextInterface* ctx) {
  g_coreml_ctx_registry.store(ctx);
}

void setModelCacheDirectory(std::string path) {
  auto p = g_coreml_ctx_registry.load();
  if (p) {
    p->setModelCacheDirectory(std::move(path));
