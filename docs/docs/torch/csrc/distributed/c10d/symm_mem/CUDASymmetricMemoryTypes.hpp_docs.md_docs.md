namespace c10d::symmetric_memory {

constexpr int symm_max_nblocks = 32;

// channels. Each signal is 32 bits, which is the minimum unit for atomic cas.

    symm_max_nblocks * max_cuda_p2p_domain_size * sizeof(uint32_t);
