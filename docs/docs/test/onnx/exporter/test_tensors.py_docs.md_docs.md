class SymbolicTensorTest(common_utils.TestCase):

        tensor = _tensors.SymbolicTensor(
            opset=onnxscript.values.Opset(domain="test", version=1)
        )
        self.assertEqual(hash(tensor), hash(tensor))
        self.assertIn(tensor, {tensor})
