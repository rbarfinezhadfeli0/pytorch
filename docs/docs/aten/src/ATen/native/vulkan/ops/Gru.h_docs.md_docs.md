class GruPackedContext final : virtual public VulkanPackedContext,
                               public torch::jit::CustomClassHolder {

  GruPackedContext(

    static constexpr uint32_t BatchFirst = 6u;

  };

   * Assigns a name to each index in the packed list.

  struct Packed final {
