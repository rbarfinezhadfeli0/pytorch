VERSION_STRING="2.9.0.dev${COMMIT_DATE}+cpu"
git rev-parse HEAD > /tmp/orig_work.txt
git reset --hard $COMMIT
USE_NIGHTLY=$VERSION_STRING python -m pip install --no-build-isolation -v -e .
