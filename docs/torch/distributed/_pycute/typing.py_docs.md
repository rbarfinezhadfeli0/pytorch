from abc import ABC


class Integer(ABC):  # noqa: B024  # Uses __subclasshook__ instead of abstract methods
    @classmethod
    def __subclasshook__(cls, c: type) -> bool:
        if c in [bool, float]:
            return False

        return issubclass(c, int)
