    const float value);

    const float* src,

    const T* src,
    float* dst,
    size_t count = 8);

TORCH_API DST_T requantize_val(double /*src_scale*/, int64_t /*src_zero_point*/, double /*dst_scale*/, int64_t /*dst_zero_point*/, SRC_T src);
