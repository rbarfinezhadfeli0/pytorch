  return a > b ? a : b;
}

inline static size_t doz(size_t a, size_t b) {
  return a < b ? 0 : a - b;
}

inline static size_t divide_round_up(size_t n, size_t q) {
  return n % q == 0 ? n / q : n / q + 1;
}

inline static size_t round_up(size_t n, size_t q) {
  return divide_round_up(n, q) * q;
