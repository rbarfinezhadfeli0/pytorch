template <typename scalar_t>
static scalar_t* conditional_data_ptr(const Tensor& t) {
  if constexpr (std::is_const_v<scalar_t>) {
    return t.defined() ? t.contiguous().const_data_ptr<scalar_t>()
                      : nullptr;
  } else {

                      : nullptr;
  }
}
